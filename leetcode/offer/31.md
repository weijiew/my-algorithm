# 剑指 Offer 30. 包含min函数的栈


栈的每一层都保存着平行层的最小值。

```cpp
class MinStack {
public:

    stack<int> s1;
    stack<int> s2;

    /** initialize your data structure here. */
    MinStack() {

    }
    
    void push(int x) {
        s1.push(x);
        if (s2.empty() || x < s2.top()) {
            s2.push(x);
        }else {
            s2.push(s2.top());
        }
    }
    
    void pop() {
        s1.pop();
        s2.pop();
    }
    
    int top() {
        return s1.top();
    }
    
    int min() {
        return s2.top();
    }
};
```
