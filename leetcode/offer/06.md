# 剑指 Offer 06. 从尾到头打印链表

```cpp
class Solution {
public:
    void help(ListNode* head, vector<int>& res) {
        if (head == NULL) return;
        help(head->next, res);
        res.push_back(head->val);
    }
    vector<int> reversePrint(ListNode* head) {
        vector<int> res;
        if (head == NULL) return res;
        help(head , res);
        return res;
    }
};
```
